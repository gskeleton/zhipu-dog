# **Watchdogs Documentation**

**Important Note**: All documentation content on this site is presented in **English** according to the original source from the repository. However, the system can accept and respond to *queries* or requests in various languages (including Indonesian).

---

**System Identity**: You are a **static, read-only documentation website** for the **Watchdogs** project.

Appearance, behavior, and tone MUST closely resemble: **[https://git-scm.com/](https://git-scm.com/)**. This is **not** a modern web application. It prioritizes **documentation**, minimalism, utilitarianism, and is intentionally simple.

> If it looks flashy, it's wrong.

---

## **1. Visual Style & Interaction (MANDATORY)**

### UI Rules (Strict)
* Layout with **navigation on the left** (sidebar)
* Background **hex color: #2a2a2a**
* **Font family: "Palatino Linotype", "Book Antiqua", Palatino, Georgia, serif**
* Font-size: 13px for body, 16px for heading
* Line-height: 1.5
* No cards, panels, or animations
* No hero sections
* No dashboard

### Content Tone
All informational messages should feel like:
* Git Manual
* Mailing list discussions
* Kernel documentation

Example tone:
```
This section explains the Linux build process.
No automation is performed.
```

MUST NOT:
* Emoji
* Marketing language
* Chatbot-like conversational tone

---

## **2. Core Behavior**
You are a **documentation viewer**.

You:
* Fetch documentation from GitHub
* Validate it
* Render it correctly

---

        // Route mapping
        const routes = {
            '#home': 'https://raw.githubusercontent.com/gskeleton/watchdogs/refs/heads/main/README.md',
            '#linux': 'https://raw.githubusercontent.com/gskeleton/watchdogs/refs/heads/main/README.md#linux',

        // Simple markdown parser (NO SYNTAX HIGHLIGHTING)
        function parseMarkdown(text) {
            // Extract code blocks first and replace with placeholders
            const codeBlocks = [];
            let blockIndex = 0;
            
            // Handle code blocks
            text = text.replace(/```(\w+)?\n([\s\S]*?)```/g, (match, lang, code) => {
                codeBlocks.push(code);
                const placeholder = `__CODE_BLOCK_${blockIndex++}__`;
                return placeholder;
            });
            
            // Also handle inline code blocks
            text = text.replace(/```([\s\S]*?)```/g, (match, code) => {
                codeBlocks.push(code);
                const placeholder = `__CODE_BLOCK_${blockIndex++}__`;
                return placeholder;
            });
            
            // Process escapes
            text = text.replace(/\\([\\`*_{}\[\]()#+\-.!])/g, '$1');
            
            // Headings
            text = text.replace(/^### (.*?)$/gm, '<h3>$1</h3>');
            text = text.replace(/^## (.*?)$/gm, '<h2>$1</h2>');
            text = text.replace(/^# (.*?)$/gm, '<h1>$1</h1>');
            
            // Bold/Italic
            text = text.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>');
            text = text.replace(/\*(.*?)\*/g, '<em>$1</em>');
            
            // Inline code
            text = text.replace(/`([^`]+)`/g, '<code>$1</code>');
            
            // Lists
            text = text.replace(/^\s*[-*+] (.*?)$/gm, '<li>$1</li>');
            text = text.replace(/^\s*\d+\. (.*?)$/gm, '<li>$1</li>');
            
            // Handle list wrapping
            text = text.replace(/(<li>.*<\/li>)(?!\s*<li>)/gs, '<ul>$1</ul>');
            
            // Links - proper format only
            text = text.replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" target="_blank">$1</a>');
            
            // Bare URLs
            text = text.replace(/(https?:\/\/[^\s<>]+[^\s<>\.])/g, '<a href="$1" target="_blank">$1</a>');
            
            // IGNORE broken image links - remove them completely
            text = text.replace(/!\[.*?\]\([^)]*\)/g, '');
            text = text.replace(/!https?:\/\/[^\s>]+[^<\n]*/g, '');
            
            // Tables (basic support)
            const lines = text.split('\n');
            let inTable = false;
            let tableRows = [];
            let result = [];
            
            for (let line of lines) {
                if (line.includes('|') && line.trim().startsWith('|')) {
                    if (!inTable) {
                        inTable = true;
                        tableRows = [];
                    }
                    tableRows.push(line);
                } else {
                    if (inTable && tableRows.length > 0) {
                        result.push(renderTable(tableRows));
                        inTable = false;
                        tableRows = [];
                    }
                    result.push(line);
                }
            }
            
            if (inTable && tableRows.length > 0) {
                result.push(renderTable(tableRows));
            }
            
            text = result.join('\n');
            
            // Paragraphs
            text = text.split('\n\n').map(p => {
                p = p.trim();
                if (p && !p.startsWith('<') && !p.endsWith('>')) {
                    return '<p>' + p + '</p>';
                }
                return p;
            }).join('\n');
            
            // Restore code blocks
            for (let i = 0; i < codeBlocks.length; i++) {
                const placeholder = `__CODE_BLOCK_${i}__`;
                const code = codeBlocks[i].replace(/</g, '&lt;').replace(/>/g, '&gt;');
                text = text.replace(placeholder, `<pre><code>${code}</code></pre>`);
            }
            
            // Clean up
            text = text.replace(/<p><\/p>/g, '');
            text = text.replace(/\n/g, ' ');
            
            return text;
        }
        
        function renderTable(rows) {
            let html = '<table>';
            let isHeader = true;
            
            for (let row of rows) {
                const cells = row.split('|').filter(cell => cell.trim() !== '');
                
                if (cells.length === 0) continue;
                
                if (row.includes('---') || row.includes('===')) {
                    isHeader = false;
                    continue;
                }
                
                html += '<tr>';
                for (let cell of cells) {
                    const tag = isHeader ? 'th' : 'td';
                    html += `<${tag}>${cell.trim()}</${tag}>`;
                }
                html += '</tr>';
            }
            
            html += '</table>';
            return html;
        }
        
        // Extract section from README
        function extractSection(text, sectionName) {
            const lines = text.split('\n');
            let inSection = false;
            let sectionLines = [];
            let found = false;
            
            for (let i = 0; i < lines.length; i++) {
                const line = lines[i];
                
                // Look for section header
                if (line.toLowerCase().includes(`## ${sectionName}`) || 
                    line.toLowerCase().includes(`# ${sectionName}`)) {
                    inSection = true;
                    found = true;
                    continue;
                }
                
                // Stop at next major section
                if (inSection && (line.startsWith('# ') || line.startsWith('## '))) {
                    break;
                }
                
                if (inSection) {
                    sectionLines.push(line);
                }
            }
            
            if (!found && sectionName === 'linux') {
                // Try alternative patterns
                for (let i = 0; i < lines.length; i++) {
                    if (lines[i].toLowerCase().includes('linux') && 
                        (lines[i].startsWith('#') || lines[i].startsWith('##'))) {
                        inSection = true;
                        found = true;
                        continue;
                    }
                    
                    if (inSection && (lines[i].startsWith('# ') || lines[i].startsWith('## '))) {
                        break;
                    }
                    
                    if (inSection) {
                        sectionLines.push(lines[i]);
                    }
                }
            }
            
            if (!found) {
                return `[ERROR] Section "${sectionName}" not found in README.`;
            }
            
            return sectionLines.join('\n');
        }
        
        // Load content
        async function loadContent() {
            const hash = window.location.hash || '#home';
            const contentDiv = document.getElementById('content');
            
            // Update active nav link
            document.querySelectorAll('.nav-link').forEach(link => {
                link.classList.remove('active');
                if (link.getAttribute('href') === hash) {
                    link.classList.add('active');
                }
            });
            
            const url = routes[hash];
            if (!url) {
                contentDiv.innerHTML = '<p>[ERROR] Route not recognized.</p>';
                return;
            }
            
            try {
                const response = await fetch(url);
                
                if (!response.ok) {
                    throw new Error(`HTTP ${response.status}: ${response.statusText}`);
                }
                
                let text = await response.text();
                
                if (!text || text.trim().length === 0) {
                    throw new Error('Empty content');
                }
                
                // Process based on route
                if (hash === '#linux') {
                    text = extractSection(text, 'linux');
                } else if (hash === '#windows-native') {
                    text = extractSection(text, 'native');
                }
                
                // Parse and render
                contentDiv.innerHTML = parseMarkdown(text);
                
            } catch (error) {
                console.error('Load error:', error);
                contentDiv.innerHTML = `
                    <p>[ERROR] Failed to load documentation.</p>
                    <p><strong>URL:</strong> ${url}</p>
                    <p><strong>Error:</strong> ${error.message}</p>
                    <p><a href="${url}" target="_blank">View raw file on GitHub</a></p>
                `;
            }
        }
        
        // Initialize
        window.addEventListener('hashchange', loadContent);
        window.addEventListener('load', () => {
            if (!window.location.hash) {
                window.location.hash = '#home';
            } else {
                loadContent();
            }
        });
        
        // Handle nav clicks
        document.addEventListener('click', (e) => {
            if (e.target.classList.contains('nav-link')) {
                e.preventDefault();
                const hash = e.target.getAttribute('href');
                window.location.hash = hash;
            }
        });
    </script>
